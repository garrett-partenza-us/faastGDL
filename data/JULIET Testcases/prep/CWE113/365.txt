
    public void bad(HttpServletRequest request, HttpServletResponse response) throws Throwable {
        String data;
        if (IO.staticTrue) {
            
            data = request.getParameter("name");
        }
        else {
            
            data = null;
        }

        if (IO.staticTrue) {
            if (data != null) {
                
                response.setHeader("Location", "/author.jsp?lang=" + data);
            }
        }
    }
