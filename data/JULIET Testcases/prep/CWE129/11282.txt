
    private void goodB2G() throws Throwable {
        int data;

        while (true) {
            
            data = -1;
            break;
        }

        while (true) {
            
            int array[] = { 0, 1, 2, 3, 4 };
            
            if (data >= 0 && data < array.length) {
                array[data] = 42;
            }
            else {
                IO.writeLine("Array index out of bounds");
            }
            break;
        }
    }
