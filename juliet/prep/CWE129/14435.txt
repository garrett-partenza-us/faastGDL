
    public void goodB2GSink(int data ) throws Throwable {

        
        int array[] = { 0, 1, 2, 3, 4 };

        
        if (data >= 0 && data < array.length) {
            IO.writeLine(array[data]);
        }
        else {
            IO.writeLine("Array index out of bounds");
        }

    }
