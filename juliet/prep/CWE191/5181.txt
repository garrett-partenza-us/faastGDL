
    private void goodB2G() throws Throwable {
        int data;

        while (true) {
            
            data = Integer.MIN_VALUE;
            break;
        }

        while (true) {
            if(data < 0) {
                
                if (data > (Integer.MIN_VALUE/2)) {
                    int result = (int)(data * 2);
                    IO.writeLine("result: " + result);
                }
                else {
                    IO.writeLine("data value is too small to perform multiplication.");
                }
            }
            break;
        }
    }
